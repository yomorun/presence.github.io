## Getting Started

Install with npm:

```
$ npm i --save @yomo/group-hug-react
```

Create a [Presence](/interface/presence) instance, Then add `<GroupHug />` to pages:

```js {30-36}
import Presence from '@yomo/presence';
import { GroupHug } from '@yomo/group-hug-react';

const id = 'YOUR_ID';
const name = 'YOUR_NAME';
const avatar = 'YOUR_AVATAR_URL';

const App = () => {
  const presence = useRef(null);
  useEffect(() => {
    const p = new Presence('https://lo.yomo.dev', {
      publicKey: process.env.NEXT_PUBLIC_PRESENCE_PUBLIC_KEY,
      id: 'client-id',
      appId: process.env.NEXT_PUBLIC_APP_ID,
      debug: true,
    });

    p.on('connected', () => {
      console.log('Connected to server: ', p.host);
      presence.current = p;
    });

    return () => {
      p.disconnect();
      presence.current = null;
    }
  }, []);
  return (
    <div>
      <GroupHug
        presence={presence}
        id={id}
        name={name}
        avatar={avatar}
        darkMode={true}
      />
    </div>
  );
};
```
