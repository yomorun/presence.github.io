## Getting Started

Install with npm:

```
$ npm i --save @yomo/group-hug-react
```

Create a [Presence](/interface/presence) instance, Then add `<GroupHug />` to pages:

```js {30-37}
import GroupHug from "@yomo/group-hug-react";
import "@yomo/group-hug-react/style.css";
import { createPresence } from "@yomo/presence";
import { nanoid } from 'nanoid';
import { useRouter } from "next/router";
import { useEffect, useRef, useState } from "react";

export default () => {
  const id = useRef<string>(nanoid());
  const [name, setName] = useState<string>("");
  const presence = useRef(null);
  const router = useRouter();

  useEffect(() => {
    if (!presence.current) {
      presence.current = createPresence("https://lo.yomo.dev/v1", {
        publicKey: process.env.NEXT_PUBLIC_PRESENCE_PUBLIC_KEY,
        id: id.current,
      });
    }
    return () => {
      presence.current = null;
    };
  }, []);

  if (!presence.current) return <div></div>;

  return (
    <div>
      <GroupHug
        presence={presence.current}
        channel="pjs-doc"
        id={id.current}
        size={36}
        darkMode={true}
        avatarBorderWidth={2}
      />
    </div>
  );
};
```
